399                        throw DbException.get(ErrorCode.CONNECTION_BROKEN_1, e, e + ": " + s);
400                    }

402                }
403            }
404            checkClosed();
405            if (switchOffCluster) {

408            checkClusterDisableAutoCommit(serverList);
409        } catch (DbException e) {

412        }
413    }

442    /**
443     * Automatically re-connect if necessary and if configured to do so.
444     *
445     * @param count the retry count index
446     * @return true if reconnected
447     */
448    private boolean autoReconnect(int count) {
449        if (!isClosed()) {
450            return false;
451        }
452        if (!autoReconnect) {
453            return false;
454        }
455        if (!cluster && !autoCommit) {
456            return false;
457        }
458        if (count > SysProperties.MAX_RECONNECT) {
459            return false;
460        }
461        lastReconnect++;
462        while (true) {
463            try {
464                embedded = connectEmbeddedOrServer(false);
465                break;
466            } catch (DbException e) {
467                if (e.getErrorCode() != ErrorCode.DATABASE_IS_IN_EXCLUSIVE_MODE) {
468                    throw e;
469                }
470                // exclusive mode: re-try endlessly
471                try {
472                    Thread.sleep(500);
473                } catch (Exception e2) {
474                    // ignore
475                }
476            }
477        }
478        if (embedded == this) {
479            // connected to a server somewhere else
480            embedded = null;
481        } else {
482            // opened an embedded connection now -
483            // must connect to this database in server mode
484            // unfortunately
485            connectEmbeddedOrServer(true);
486        }
487        recreateSessionState();
488        if (eventListener != null) {
489            eventListener.setProgress(DatabaseEventListener.STATE_RECONNECTED, databaseName, count,
490                    SysProperties.MAX_RECONNECT);
491        }
492        return true;
493    }

548    /**
549     * Called to flush the output after data has been sent to the server and
550     * just before receiving data. This method also reads the status code from
551     * the server and throws any exception the server sent.
552     *
553     * @param transfer the transfer object
554     * @throws DbException if the server sent an exception
555     * @throws IOException if there is a communication problem between client
556     *             and server
557     */
558    public void done(Transfer transfer) throws IOException {
559        transfer.flush();
560        int status = transfer.readInt();
561        if (status == STATUS_ERROR) {
562            String sqlstate = transfer.readString();
563            String message = transfer.readString();
564            String sql = transfer.readString();
565            int errorCode = transfer.readInt();
566            String stackTrace = transfer.readString();
567            JdbcSQLException s = new JdbcSQLException(message, sql, sqlstate, errorCode, null, stackTrace);
568            if (errorCode == ErrorCode.CONNECTION_BROKEN_1) {
569                // allow re-connect
570                IOException e = new IOException(s.toString());
571                e.initCause(s);
572                throw e;
573            }
574            throw DbException.convert(s);
575        } else if (status == STATUS_CLOSED) {
576            transferList = null;
577        } else if (status == STATUS_OK_STATE_CHANGED) {
578            sessionStateChanged = true;
579        } else if (status == STATUS_OK) {
580            // ok
581        } else {
582            throw DbException.get(ErrorCode.CONNECTION_BROKEN_1, "unexpected status " + status);
583        }
584    }

694    public synchronized int readLob(long lobId, byte[] hmac, long offset, byte[] buff, int off, int length) {
695        for (int i = 0, count = 0; i < transferList.size(); i++) {
696            Transfer transfer = transferList.get(i);
697            try {
698                traceOperation("LOB_READ", (int) lobId);
699                transfer.writeInt(SessionRemote.LOB_READ);
700                transfer.writeLong(lobId);
701                if (clientVersion >= Constants.TCP_PROTOCOL_VERSION_12) {
702                    transfer.writeBytes(hmac);
703                }
704                transfer.writeLong(offset);
705                transfer.writeInt(length);
706                done(transfer);
707                length = transfer.readInt();
708                if (length <= 0) {
709                    return length;
710                }
711                transfer.readBytes(buff, off, length);
712                return length;
713            } catch (IOException e) {
714                removeServer(e, i--, ++count);
715            }
716        }
717        return 1;
718    }
