426        for (rows.reset(); rows.hasNext();) {
427            if ((++rowScanCount & 127) == 0) {
428                prepared.checkCanceled();

431            rows.next();
432            removeRow(session, o);
433            session.log(this, UndoLogRecord.DELETE, o);
434        }
435        // add the new rows
436        for (rows.reset(); rows.hasNext();) {
437            if ((++rowScanCount & 127) == 0) {
438                prepared.checkCanceled();
439            }
440            rows.next();
441            Row n = rows.next();
442            try {

446                    session.rollbackTo(rollback, false);

448                throw e;
449            }
450            session.log(this, UndoLogRecord.INSERT, n);
451        }

498     * references and indexes are dropped.

500     * @param session the session

505    public void dropSingleColumnConstraintsAndIndexes(Session session, Column col) {
506        ArrayList<Constraint> constraintsToDrop = New.arrayList();
507        if (constraints != null) {

528                }

534                } else {
535                    throw DbException.get(ErrorCode.COLUMN_IS_REFERENCED_1, index.getSQL());
536                }
537            }
538        }
539        for (Constraint c : constraintsToDrop) {
540            session.getDatabase().removeSchemaObject(session, c);
541        }
542        for (Index i : indexesToDrop) {
543            // the index may already have been dropped when dropping the constraint
