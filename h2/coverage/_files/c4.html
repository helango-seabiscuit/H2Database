449    private void removeChild(int i) {
450        readAllRows();
451        entryCount--;
452        if (pageStoreInternalCount) {
453            updateRowCount(-index.getPage(childPageIds[i]).getRowCount());
454        }
455        written = false;
456        changeCount = index.getPageStore().getChangeCount();
457        if (entryCount < 0) {
458            DbException.throwInternalError();
459        }
460        if (entryCount > i) {
461            int startNext = i > 0 ? offsets[i - 1] : index.getPageStore().getPageSize();
462            int rowLength = startNext - offsets[i];
463            add(offsets, i, entryCount + 1, rowLength);
464        }
465        rows = remove(rows, entryCount + 1, i);
466        offsets = remove(offsets, entryCount + 1, i);
467        childPageIds = remove(childPageIds, entryCount + 2, i);
468        start -= CHILD_OFFSET_PAIR_LENGTH;

542        p2.rows = rows;
543        p2.entryCount = entryCount;

564        }
