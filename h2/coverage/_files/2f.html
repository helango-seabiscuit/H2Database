455            expand(data, lastPage);
456            byte[] d = data[lastPage];
457            for (int i = (int) (newLength & BLOCK_SIZE_MASK); i < BLOCK_SIZE; i++) {
458                d[i] = 0;
459            }
460            if (compress) {
461                compressLater(data, lastPage);
462            }
463        }
464    }

466    private void changeLength(long len) {
467        length = len;
468        len = MathUtils.roundUpLong(len, BLOCK_SIZE);
469        int blocks = (int) (len >>> BLOCK_SIZE_SHIFT);
470        if (blocks != data.length) {
471            byte[][] n = new byte[blocks][];
472            System.arraycopy(data, 0, n, 0, Math.min(data.length, n.length));
473            for (int i = data.length; i < blocks; i++) {
474                n[i] = COMPRESSED_EMPTY_BLOCK;
475            }
476            data = n;
477        }
478    }

480    /**
481     * Read or write.
482     *
483     * @param pos the position
484     * @param b the byte array
485     * @param off the offset within the byte array
486     * @param len the number of bytes
487     * @param write true for writing
488     * @return the new position
489     */
490    long readWrite(long pos, byte[] b, int off, int len, boolean write) {
491        long end = pos + len;
492        if (end > length) {
493            if (write) {
494                changeLength(end);
495            } else {
496                return pos;
497            }
498        }
499        while (len > 0) {
500            int l = (int) Math.min(len, BLOCK_SIZE - (pos & BLOCK_SIZE_MASK));
501            int page = (int) (pos >>> BLOCK_SIZE_SHIFT);
502            expand(data, page);
503            byte[] block = data[page];
504            int blockOffset = (int) (pos & BLOCK_SIZE_MASK);
505            if (write) {
506                System.arraycopy(b, off, block, blockOffset, l);
507            } else {
508                System.arraycopy(block, blockOffset, b, off, l);
509            }
510            if (compress) {
511                compressLater(data, page);
512            }
513            off += l;
514            pos += l;
515            len -= l;
516        }
517        return pos;
518    }
