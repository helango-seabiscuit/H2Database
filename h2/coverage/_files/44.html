416    private int readBuffer() throws IOException {
417        if (endOfFile) {
418            return -1;
419        }
420        int keep;
421        if (inputBufferStart >= 0) {
422            keep = inputBufferPos - inputBufferStart;
423            if (keep > 0) {
424                char[] src = inputBuffer;
425                if (keep + Constants.IO_BUFFER_SIZE > src.length) {
426                    inputBuffer = new char[src.length * 2];
427                }
428                System.arraycopy(src, inputBufferStart, inputBuffer, 0, keep);
429            }
430            inputBufferStart = 0;
431        } else {
432            keep = 0;
433        }
434        inputBufferPos = keep;
435        int len = input.read(inputBuffer, keep, Constants.IO_BUFFER_SIZE);
436        if (len == -1) {
437            // ensure bufferPos > bufferEnd
438            // even after pushBack
439            inputBufferEnd = -1024;
440            endOfFile = true;
441            // ensure the right number of characters are read
442            // in case the input buffer is still used
443            inputBufferPos++;
444            return -1;
445        }
446        inputBufferEnd = keep + len;
447        return inputBuffer[inputBufferPos++];
448    }

450    private String readValue() throws IOException {
451        endOfLine = false;
452        inputBufferStart = inputBufferPos;
453        while (true) {
454            int ch = readChar();
455            if (ch == fieldDelimiter) {
456                // delimited value
457                boolean containsEscape = false;
458                inputBufferStart = inputBufferPos;
459                int sep;
460                while (true) {
461                    ch = readChar();
462                    if (ch == fieldDelimiter) {
463                        ch = readChar();
464                        if (ch != fieldDelimiter) {
465                            sep = 2;
466                            break;
467                        }
468                        containsEscape = true;
469                    } else if (ch == escapeCharacter) {
470                        ch = readChar();
471                        if (ch < 0) {
472                            sep = 1;
473                            break;
474                        }
475                        containsEscape = true;
476                    } else if (ch < 0) {
477                        sep = 1;
478                        break;
479                    }
480                }
481                String s = new String(inputBuffer, inputBufferStart, inputBufferPos - inputBufferStart - sep);
482                if (containsEscape) {
483                    s = unEscape(s);
484                }
485                inputBufferStart = -1;
486                while (true) {
487                    if (ch == fieldSeparatorRead) {
488                        break;
489                    } else if (ch == '\n' || ch < 0 || ch == '\r') {
490                        endOfLine = true;
491                        break;
492                    } else if (ch == ' ' || ch == '\t') {
493                        // ignore
494                    } else {
495                        pushBack();
496                        break;
497                    }
498                    ch = readChar();
499                }
500                return s;
501            } else if (ch == '\n' || ch < 0 || ch == '\r') {
502                endOfLine = true;
503                return null;
504            } else if (ch == fieldSeparatorRead) {
505                // null
506                return null;
507            } else if (ch <= ' ') {
508                // ignore spaces
509                continue;
510            } else if (lineComment != 0 && ch == lineComment) {
511                // comment until end of line
512                inputBufferStart = -1;
513                while (true) {
514                    ch = readChar();
515                    if (ch == '\n' || ch < 0 || ch == '\r') {
516                        break;
517                    }
518                }
519                endOfLine = true;
520                return null;
521            } else {
522                // un-delimited value
523                while (true) {
524                    ch = readChar();
525                    if (ch == fieldSeparatorRead) {
526                        break;
527                    } else if (ch == '\n' || ch < 0 || ch == '\r') {
528                        endOfLine = true;
529                        break;
530                    }
531                }
532                String s = new String(inputBuffer, inputBufferStart, inputBufferPos - inputBufferStart - 1);
533                if (!preserveWhitespace) {
534                    s = s.trim();
535                }
536                inputBufferStart = -1;
537                // check un-delimited value for nullString
538                return readNull(s);
539            }
540        }
541    }

547    private String unEscape(String s) {
548        StringBuilder buff = new StringBuilder(s.length());
549        int start = 0;
550        char[] chars = null;
551        while (true) {
552            int idx = s.indexOf(escapeCharacter, start);
553            if (idx < 0) {
554                idx = s.indexOf(fieldDelimiter, start);
555                if (idx < 0) {
556                    break;
557                }
558            }
559            if (chars == null) {
560                chars = s.toCharArray();
561            }
562            buff.append(chars, start, idx - start);
563            if (idx == s.length() - 1) {
564                start = s.length();
565                break;
566            }
567            buff.append(chars[idx + 1]);
568            start = idx + 2;
569        }
570        buff.append(s.substring(start));
571        return buff.toString();
572    }
