894        } else if (x instanceof Date) {
895            return ValueDate.get((Date) x);
896        } else if (x instanceof Time) {
897            return ValueTime.get((Time) x);
898        } else if (x instanceof Timestamp) {
899            return ValueTimestamp.get((Timestamp) x);
900        } else if (x instanceof java.util.Date) {
901            return ValueTimestamp.get(new Timestamp(((java.util.Date) x).getTime()));
902        } else if (x instanceof java.io.Reader) {
903            Reader r = new BufferedReader((java.io.Reader) x);
904            return session.getDataHandler().getLobStorage().createClob(r, -1);
905        } else if (x instanceof java.sql.Clob) {
906            try {
907                Reader r = new BufferedReader(((java.sql.Clob) x).getCharacterStream());
908                return session.getDataHandler().getLobStorage().createClob(r, -1);
909            } catch (SQLException e) {
910                throw DbException.convert(e);
911            }
912        } else if (x instanceof java.io.InputStream) {
913            return session.getDataHandler().getLobStorage().createBlob((java.io.InputStream) x, -1);
914        } else if (x instanceof java.sql.Blob) {
915            try {
916                return session.getDataHandler().getLobStorage().createBlob(((java.sql.Blob) x).getBinaryStream(), -1);
917            } catch (SQLException e) {
918                throw DbException.convert(e);
919            }
920        } else if (x instanceof ResultSet) {
921            if (x instanceof SimpleResultSet) {
922                return ValueResultSet.get((ResultSet) x);
923            }
924            return ValueResultSet.getCopy((ResultSet) x, Integer.MAX_VALUE);
925        } else if (x instanceof UUID) {
926            UUID u = (UUID) x;
927            return ValueUuid.get(u.getMostSignificantBits(), u.getLeastSignificantBits());
928        } else if (x instanceof Object[]) {
929            // (a.getClass().isArray());
930            // (a.getClass().getComponentType().isPrimitive());
931            Object[] o = (Object[]) x;
932            int len = o.length;
933            Value[] v = new Value[len];
934            for (int i = 0; i < len; i++) {
935                v[i] = convertToValue(session, o[i], type);
936            }
937            return ValueArray.get(x.getClass().getComponentType(), v);
938        } else if (x instanceof Character) {
939            return ValueStringFixed.get(((Character) x).toString());
940        } else {
941            return ValueJavaObject.getNoCopy(x, null);
942        }
943    }
