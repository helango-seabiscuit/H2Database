515    private static byte[] loadResource(String name) throws IOException {
516        InputStream in = Utils.class.getResourceAsStream("data.zip");
517        if (in == null) {

532                if (!entryName.startsWith("/")) {
533                    entryName = "/" + entryName;
534                }
535                if (entryName.equals(name)) {
536                    ByteArrayOutputStream out = new ByteArrayOutputStream();
537                    IOUtils.copy(zipIn, out);
538                    zipIn.closeEntry();
539                    return out.toByteArray();
540                }
541                zipIn.closeEntry();
542            }

586    private static Object callMethod(
587            Object instance, Class<?> clazz,
588            String methodName,
589            Object... params) throws Exception {
590        Method best = null;
591        int bestMatch = 0;
592        boolean isStatic = instance == null;
593        for (Method m : clazz.getMethods()) {
594            if (Modifier.isStatic(m.getModifiers()) == isStatic && m.getName().equals(methodName)) {
595                int p = match(m.getParameterTypes(), params);
596                if (p > bestMatch) {
597                    bestMatch = p;
598                    best = m;
599                }
600            }
601        }
602        if (best == null) {
603            throw new NoSuchMethodException(methodName);
604        }
605        return best.invoke(instance, params);
606    }

609     * Creates a new instance. This will try to use the constructor where the
610     * most parameter classes match exactly (this algorithm is simpler than the
611     * one in the Java specification, but works well for most cases).

629        }

633    private static int match(Class<?>[] params, Object[] values) {
634        int len = params.length;
635        if (len == values.length) {
636            int points = 1;
637            for (int i = 0; i < len; i++) {
638                Class<?> pc = getNonPrimitiveClass(params[i]);
639                Object v = values[i];
640                Class<?> vc = v == null ? null : v.getClass();
641                if (pc == vc) {
642                    points++;
643                } else if (vc == null) {
644                    // can't verify
645                } else if (!pc.isAssignableFrom(vc)) {
646                    return 0;
647                }
648            }
649            return points;
650        }
651        return 0;
652    }
