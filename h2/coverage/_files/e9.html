403            sortColumns.add(exprCol.getColumn());
404        }
405        Column[] sortCols = sortColumns.toArray(new Column[sortColumns.size()]);
406        int[] sortTypes = sort.getSortTypes();
407        if (sortCols.length == 0) {
408            // sort just on constants - can use scan index
409            return topTableFilter.getTable().getScanIndex(session);
410        }
411        ArrayList<Index> list = topTableFilter.getTable().getIndexes();
412        if (list != null) {
413            for (int i = 0, size = list.size(); i < size; i++) {
414                Index index = list.get(i);
415                if (index.getCreateSQL() == null) {
416                    // can't use the scan index
417                    continue;
418                }
419                if (index.getIndexType().isHash()) {
420                    continue;
421                }
422                IndexColumn[] indexCols = index.getIndexColumns();
423                if (indexCols.length < sortCols.length) {
424                    continue;
425                }
426                boolean ok = true;
427                for (int j = 0; j < sortCols.length; j++) {
428                    // the index and the sort order must start
429                    // with the exact same columns
430                    IndexColumn idxCol = indexCols[j];
431                    Column sortCol = sortCols[j];
432                    if (idxCol.column != sortCol) {
433                        ok = false;
434                        break;
435                    }
436                    if (idxCol.sortType != sortTypes[j]) {
437                        // NULL FIRST for ascending and NULLS LAST
438                        // for descending would actually match the default
439                        ok = false;
440                        break;
441                    }
442                }
443                if (ok) {
444                    return index;
445                }
446            }
447        }
448        if (sortCols.length == 1 && sortCols[0].getColumnId() == -1) {
449            // special case: order by _ROWID_
450            Index index = topTableFilter.getTable().getScanIndex(session);
451            if (index.isRowIdIndex()) {
452                return index;
453            }
454        }
455        return null;
456    }

561            } else if (l >= 0) {

566        LocalResult result = null;

614                }

619            }
620        }

622            result.setOffset(offsetExpr.getValue(session).getInt());
623        }

626        }

632                }

635            }

637        }

785    public void prepare() {
786        if (isPrepared) {
787            // sometimes a subquery is prepared twice (CREATE TABLE AS SELECT)
788            return;
789        }
790        if (SysProperties.CHECK && !checkInit) {
791            DbException.throwInternalError("not initialized");
792        }
793        if (orderList != null) {
794            sort = prepareOrder(orderList, expressions.size());
795            orderList = null;
796        }
797        for (int i = 0; i < expressions.size(); i++) {
798            Expression e = expressions.get(i);
799            expressions.set(i, e.optimize(session));
800        }
801        if (condition != null) {
802            condition = condition.optimize(session);
803            for (TableFilter f : filters) {
804                // outer joins: must not add index conditions such as
805                // "c is null" - example:
806                // create table parent(p int primary key) as select 1;
807                // create table child(c int primary key, pc int);
808                // insert into child values(2, 1);
809                // select p, c from parent
810                // left outer join child on p = pc where c is null;
811                if (!f.isJoinOuter() && !f.isJoinOuterIndirect()) {
812                    condition.createIndexConditions(session, f);
813                }
814            }
815        }
816        if (isGroupQuery && groupIndex == null && havingIndex < 0 && filters.size() == 1) {
817            if (condition == null) {
818                Table t = filters.get(0).getTable();
819                ExpressionVisitor optimizable = ExpressionVisitor.getOptimizableVisitor(t);
820                isQuickAggregateQuery = isEverything(optimizable);
821            }
822        }
823        cost = preparePlan();
824        if (distinct && session.getDatabase().getSettings().optimizeDistinct &&
825                !isGroupQuery && filters.size() == 1 &&
826                expressions.size() == 1 && condition == null) {
827            Expression expr = expressions.get(0);
828            expr = expr.getNonAliasExpression();
829            if (expr instanceof ExpressionColumn) {
830                Column column = ((ExpressionColumn) expr).getColumn();
831                int selectivity = column.getSelectivity();
832                Index columnIndex = topTableFilter.getTable().getIndexForColumn(column, true);
833                if (columnIndex != null && selectivity != Constants.SELECTIVITY_DEFAULT && selectivity < 20) {
834                    // the first column must be ascending
835                    boolean ascending = columnIndex.getIndexColumns()[0].sortType == SortOrder.ASCENDING;
836                    Index current = topTableFilter.getIndex();
837                    // if another index is faster
838                    if (columnIndex.canFindNext() && ascending &&
839                            (current == null || current.getIndexType().isScan() || columnIndex == current)) {
840                        IndexType type = columnIndex.getIndexType();
841                        // hash indexes don't work, and unique single column indexes don't work
842                        if (!type.isHash() && (!type.isUnique() || columnIndex.getColumns().length > 1)) {
843                            topTableFilter.setIndex(columnIndex);
844                            isDistinctQuery = true;
845                        }
846                    }
847                }
848            }
849        }
850        if (sort != null && !isQuickAggregateQuery && !isGroupQuery) {
851            Index index = getSortIndex();
852            if (index != null) {
853                Index current = topTableFilter.getIndex();
854                if (current.getIndexType().isScan() || current == index) {
855                    topTableFilter.setIndex(index);
856                    if (!topTableFilter.hasInComparisons()) {
857                        // in(select ...) and in(1,2,3) my return the key in another order
858                        sortUsingIndex = true;
859                    }
860                } else if (index.getIndexColumns().length >= current.getIndexColumns().length) {
861                    IndexColumn[] sortColumns = index.getIndexColumns();
862                    IndexColumn[] currentColumns = current.getIndexColumns();
863                    boolean swapIndex = false;
864                    for (int i = 0; i < currentColumns.length; i++) {
865                        if (sortColumns[i].column != currentColumns[i].column) {
866                            swapIndex = false;
867                            break;
868                        }
869                        if (sortColumns[i].sortType != currentColumns[i].sortType) {
870                            swapIndex = true;
871                        }
872                    }
873                    if (swapIndex) {
874                        topTableFilter.setIndex(index);
875                        sortUsingIndex = true;
876                    }
877                }
878            }
879        }
880        if (!isQuickAggregateQuery && isGroupQuery && getGroupByExpressionCount() > 0) {
881            Index index = getGroupSortedIndex();
882            Index current = topTableFilter.getIndex();
883            if (index != null && (current.getIndexType().isScan() || current == index)) {
884                topTableFilter.setIndex(index);
885                isGroupSortedQuery = true;
886            }
887        }
