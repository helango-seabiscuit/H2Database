466    /**
467     * Shrink the file so there are no empty pages at the end.
468     *
469     * @param compactMode 0 if no compacting should happen, otherwise
470     * TransactionCommand.SHUTDOWN_COMPACT or TransactionCommand.SHUTDOWN_DEFRAG
471     */
472    public synchronized void compact(int compactMode) {
473        if (!database.getSettings().pageStoreTrim) {
474            return;
475        }
476        if (SysProperties.MODIFY_ON_WRITE && readMode && compactMode == 0) {
477            return;
478        }
479        openForWriting();
480        // find the last used page
481        int lastUsed = -1;
482        for (int i = getFreeListId(pageCount); i >= 0; i--) {
483            lastUsed = getFreeList(i).getLastUsed();
484            if (lastUsed != -1) {
485                break;
486            }
487        }
488        // open a new log at the very end
489        // (to be truncated later)
490        writeBack();
491        log.free();
492        recoveryRunning = true;
493        try {
494            logFirstTrunkPage = lastUsed + 1;
495            allocatePage(logFirstTrunkPage);
496            log.openForWriting(logFirstTrunkPage, true);
497            // ensure the free list is backed up again
498            log.checkpoint();
499        } finally {
500            recoveryRunning = false;
501        }
502        long start = System.currentTimeMillis();
503        boolean isCompactFully = compactMode == CommandInterface.SHUTDOWN_COMPACT;
504        boolean isDefrag = compactMode == CommandInterface.SHUTDOWN_DEFRAG;
505 
506        if (database.getSettings().defragAlways) {
507            isCompactFully = isDefrag = true;
508        }
509 
510        int maxCompactTime = database.getSettings().maxCompactTime;
511        int maxMove = database.getSettings().maxCompactCount;
512 
513        if (isCompactFully || isDefrag) {
514            maxCompactTime = Integer.MAX_VALUE;
515            maxMove = Integer.MAX_VALUE;
516        }
517        int blockSize = isCompactFully ? COMPACT_BLOCK_SIZE : 1;
518        int firstFree = MIN_PAGE_COUNT;
519        for (int x = lastUsed, j = 0; x > MIN_PAGE_COUNT && j < maxMove; x -= blockSize) {
520            for (int full = x - blockSize + 1; full <= x; full++) {
521                if (full > MIN_PAGE_COUNT && isUsed(full)) {
522                    synchronized (this) {
523                        firstFree = getFirstFree(firstFree);
524                        if (firstFree == -1 || firstFree >= full) {
525                            j = maxMove;
526                            break;
527                        }
528                        if (compact(full, firstFree)) {
529                            j++;
530                            long now = System.currentTimeMillis();
531                            if (now > start + maxCompactTime) {
532                                j = maxMove;
533                                break;
534                            }
535                        }
536                    }
537                }
538            }
539        }
540        if (isDefrag) {
541            log.checkpoint();
542            writeBack();
543            cache.clear();
544            ArrayList<Table> tables = database.getAllTablesAndViews(false);
545            recordedPagesList = New.arrayList();
546            recordedPagesIndex = new IntIntHashMap();
547            recordPageReads = true;
548            Session s = database.getSystemSession();
549            for (Table table : tables) {
550                if (!table.isTemporary() && Table.TABLE.equals(table.getTableType())) {
551                    Index scanIndex = table.getScanIndex(s);
552                    Cursor cursor = scanIndex.find(s, null, null);
553                    while (cursor.next()) {
554                        cursor.get();
555                    }
556                    for (Index index : table.getIndexes()) {
557                        if (index != scanIndex && index.canScan()) {
558                            cursor = index.find(s, null, null);
559                            while (cursor.next()) {
560                                // the data is already read
561                            }
562                        }
563                    }
564                }
565            }
566            recordPageReads = false;
567            int target = MIN_PAGE_COUNT - 1;
568            int temp = 0;
569            for (int i = 0, size = recordedPagesList.size(); i < size; i++) {
570                log.checkpoint();
571                writeBack();
572                int source = recordedPagesList.get(i);
573                Page pageSource = getPage(source);
574                if (!pageSource.canMove()) {
575                    continue;
576                }
577                while (true) {
578                    Page pageTarget = getPage(++target);
579                    if (pageTarget == null || pageTarget.canMove()) {
580                        break;
581                    }
582                }
583                if (target == source) {
584                    continue;
585                }
586                temp = getFirstFree(temp);
587                if (temp == -1) {
588                    DbException.throwInternalError("no free page for defrag");
589                }
590                cache.clear();
591                swap(source, target, temp);
592                int index = recordedPagesIndex.get(target);
593                if (index != IntIntHashMap.NOT_FOUND) {
594                    recordedPagesList.set(index, source);
595                    recordedPagesIndex.put(source, index);
596                }
597                recordedPagesList.set(i, target);
598                recordedPagesIndex.put(target, i);
599            }
600            recordedPagesList = null;
601            recordedPagesIndex = null;
602        }
603        // TODO can most likely be simplified
604        checkpoint();
605        log.checkpoint();
606        writeIndexRowCounts();
607        log.checkpoint();
608        writeBack();
609        commit(systemSession);
610        writeBack();
611        log.checkpoint();
612 
613        log.free();
614        // truncate the log
615        recoveryRunning = true;
616        try {
617            setLogFirstPage(++logKey, 0, 0);
618        } finally {
619            recoveryRunning = false;
620        }
621        writeBack();
622        for (int i = getFreeListId(pageCount); i >= 0; i--) {
623            lastUsed = getFreeList(i).getLastUsed();
624            if (lastUsed != -1) {
625                break;
626            }
627        }
628        int newPageCount = lastUsed + 1;
629        if (newPageCount < pageCount) {
630            freed.set(newPageCount, pageCount, false);
631        }
632        pageCount = newPageCount;
633        // the easiest way to remove superfluous entries
634        freeLists.clear();
635        trace.debug("pageCount: " + pageCount);
636        long newLength = (long) pageCount << pageSizeShift;
637        if (file.length() != newLength) {
638            file.setLength(newLength);
639            writeCount++;
640        }
641    }

722    /**
723     * Read a page from the store.
724     *
725     * @param pageId the page id
726     * @return the page
727     */
728    public synchronized Page getPage(int pageId) {
729        Page p = (Page) cache.get(pageId);
730        if (p != null) {
731            return p;
732        }
733 
734        Data data = createData();
735        readPage(pageId, data);
736        int type = data.readByte();
737        if (type == Page.TYPE_EMPTY) {
738            return null;
739        }
740        data.readShortInt();
741        data.readInt();
742        if (!checksumTest(data.getBytes(), pageId, pageSize)) {
743            throw DbException.get(ErrorCode.FILE_CORRUPTED_1, "wrong checksum");
744        }
745        switch (type & ~Page.FLAG_LAST) {
746        case Page.TYPE_FREE_LIST:
747            p = PageFreeList.read(this, data, pageId);
748            break;
749        case Page.TYPE_DATA_LEAF: {
750            int indexId = data.readVarInt();
751            PageIndex idx = metaObjects.get(indexId);
752            if (idx == null) {
753                throw DbException.get(ErrorCode.FILE_CORRUPTED_1, "index not found " + indexId);
754            }
755            if (!(idx instanceof PageDataIndex)) {
756                throw DbException.get(ErrorCode.FILE_CORRUPTED_1, "not a data index " + indexId + " " + idx);
757            }
758            PageDataIndex index = (PageDataIndex) idx;
759            if (statistics != null) {
760                statisticsIncrement(index.getTable().getName() + "." + index.getName() + " read");
761            }
762            p = PageDataLeaf.read(index, data, pageId);
763            break;
764        }
765        case Page.TYPE_DATA_NODE: {
766            int indexId = data.readVarInt();
767            PageIndex idx = metaObjects.get(indexId);
768            if (idx == null) {
769                throw DbException.get(ErrorCode.FILE_CORRUPTED_1, "index not found " + indexId);
770            }
771            if (!(idx instanceof PageDataIndex)) {
772                throw DbException.get(ErrorCode.FILE_CORRUPTED_1, "not a data index " + indexId + " " + idx);
773            }
774            PageDataIndex index = (PageDataIndex) idx;
775            if (statistics != null) {
776                statisticsIncrement(index.getTable().getName() + "." + index.getName() + " read");
777            }
778            p = PageDataNode.read(index, data, pageId);
779            break;
780        }
781        case Page.TYPE_DATA_OVERFLOW: {
782            p = PageDataOverflow.read(this, data, pageId);
783            if (statistics != null) {
784                statisticsIncrement("overflow read");
785            }
786            break;
787        }
788        case Page.TYPE_BTREE_LEAF: {
789            int indexId = data.readVarInt();
790            PageIndex idx = metaObjects.get(indexId);
791            if (idx == null) {
792                throw DbException.get(ErrorCode.FILE_CORRUPTED_1, "index not found " + indexId);
