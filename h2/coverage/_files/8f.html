727     *
728     * @param lobId the lob id
729     * @param seq the sequence number
730     * @param pos the position within the lob
731     * @param b the data
732     * @param compressAlgorithm the compression algorithm (may be null)
733     */
734    void storeBlock(long lobId, int seq, long pos, byte[] b, String compressAlgorithm) throws SQLException {
735        long block;
736        boolean blockExists = false;
737        if (compressAlgorithm != null) {
738            b = compress.compress(b, compressAlgorithm);
739        }
740        int hash = Arrays.hashCode(b);
741        synchronized (handler) {
742            block = getHashCacheBlock(hash);
743            if (block != -1) {
744                String sql =  "SELECT COMPRESSED, DATA FROM " + LOB_DATA +
745                        " WHERE BLOCK = ?";
746                PreparedStatement prep = prepare(sql);
747                prep.setLong(1, block);
748                ResultSet rs = prep.executeQuery();
749                if (rs.next()) {
750                    boolean compressed = rs.getInt(1) != 0;
751                    byte[] compare = rs.getBytes(2);
752                    if (compressed == (compressAlgorithm != null) && Arrays.equals(b, compare)) {
753                        blockExists = true;
754                    }
755                }
756                reuse(sql, prep);
757            }
758            if (!blockExists) {
759                block = nextBlock++;
760                setHashCacheBlock(hash, block);
761                String sql = "INSERT INTO " + LOB_DATA + "(BLOCK, COMPRESSED, DATA) VALUES(?, ?, ?)";
762                PreparedStatement prep = prepare(sql);
763                prep.setLong(1, block);
764                prep.setInt(2, compressAlgorithm == null ? 0 : 1);
765                prep.setBytes(3, b);
766                prep.execute();
767                reuse(sql, prep);
768            }
769            String sql = "INSERT INTO " + LOB_MAP + "(LOB, SEQ, POS, HASH, BLOCK) VALUES(?, ?, ?, ?, ?)";
770            PreparedStatement prep = prepare(sql);
771            prep.setLong(1, lobId);
772            prep.setInt(2, seq);
773            prep.setLong(3, pos);
774            prep.setLong(4, hash);
775            prep.setLong(5, block);
776            prep.execute();
777            reuse(sql, prep);
778        }
