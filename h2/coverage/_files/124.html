529     * Add a check constraint expression to this column. An existing check
530     * constraint constraint is added using AND.
531     *
532     * @param session the session
533     * @param expr the (additional) constraint
534     */
535    public void addCheckConstraint(Session session, Expression expr) {
536        resolver = new SingleColumnResolver(this);
537        synchronized (this) {
538            String oldName = name;
539            if (name == null) {
540                name = "VALUE";
541            }
542            expr.mapColumns(resolver, 0);
543            name = oldName;
544        }
545        expr = expr.optimize(session);
546        resolver.setValue(ValueNull.INSTANCE);
547        // check if the column is mapped
548        synchronized (this) {
549            expr.getValue(session);
550        }
551        if (checkConstraint == null) {
552            checkConstraint = expr;
553        } else {
554            checkConstraint = new ConditionAndOr(ConditionAndOr.AND, checkConstraint, expr);
555        }
556        checkConstraintSQL = getCheckConstraintSQL(session, name);
557    }

564     * @return the constraint expression
565     */
566    public Expression getCheckConstraint(Session session, String asColumnName) {
567        if (checkConstraint == null) {
568            return null;
569        }
