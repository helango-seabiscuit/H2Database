411    public void lock(Session session, boolean exclusive, boolean force) {
412        int lockMode = database.getLockMode();
413        if (lockMode == Constants.LOCK_MODE_OFF) {
414            return;
415        }
416        if (!force && database.isMultiVersion()) {
417            // MVCC: update, delete, and insert use a shared lock.
418            // Select doesn't lock except when using FOR UPDATE and
419            // the system property h2.selectForUpdateMvcc
420            // is not enabled
421            if (exclusive) {
422                exclusive = false;
423            } else {
424                if (lockExclusive == null) {
425                    return;
426                }
427            }
428        }
429        if (lockExclusive == session) {
430            return;
431        }
432        synchronized (database) {
433            try {
434                doLock(session, lockMode, exclusive);
435            } finally {
436                session.setWaitForLock(null);
437            }
438        }
439    }

441    private void doLock(Session session, int lockMode, boolean exclusive) {
442        traceLock(session, exclusive, "requesting for");
443        // don't get the current time unless necessary
444        long max = 0;
445        boolean checkDeadlock = false;
446        while (true) {
447            if (lockExclusive == session) {
448                return;
449            }
450            if (exclusive) {
451                if (lockExclusive == null) {
452                    if (lockShared.isEmpty()) {
453                        traceLock(session, exclusive, "added for");
454                        session.addLock(this);
455                        lockExclusive = session;
456                        return;
457                    } else if (lockShared.size() == 1 && lockShared.contains(session)) {
458                        traceLock(session, exclusive, "add (upgraded) for ");
459                        lockExclusive = session;
460                        return;
461                    }
462                }
463            } else {
464                if (lockExclusive == null) {
465                    if (lockMode == Constants.LOCK_MODE_READ_COMMITTED) {
466                        if (!database.isMultiThreaded() && !database.isMultiVersion()) {
467                            // READ_COMMITTED: a read lock is acquired,
468                            // but released immediately after the operation
469                            // is complete.
470                            // When allowing only one thread, no lock is
471                            // required.
472                            // Row level locks work like read committed.
473                            return;
474                        }
475                    }
476                    if (!lockShared.contains(session)) {
477                        traceLock(session, exclusive, "ok");
478                        session.addLock(this);
479                        lockShared.add(session);
480                    }
481                    return;
482                }
483            }
484            session.setWaitForLock(this);
485            if (checkDeadlock) {
486                ArrayList<Session> sessions = checkDeadlock(session, null, null);
487                if (sessions != null) {
488                    throw DbException.get(ErrorCode.DEADLOCK_1, getDeadlockDetails(sessions));
489                }
490            } else {
491                // check for deadlocks from now on
492                checkDeadlock = true;
493            }
494            long now = System.currentTimeMillis();
495            if (max == 0) {
496                // try at least one more time
497                max = now + session.getLockTimeout();
498            } else if (now >= max) {
499                traceLock(session, exclusive, "timeout after " + session.getLockTimeout());
500                throw DbException.get(ErrorCode.LOCK_TIMEOUT_1, getName());
501            }
502            try {
503                traceLock(session, exclusive, "waiting for");
504                if (database.getLockMode() == Constants.LOCK_MODE_TABLE_GC) {
505                    for (int i = 0; i < 20; i++) {
506                        long free = Runtime.getRuntime().freeMemory();
507                        System.gc();
508                        long free2 = Runtime.getRuntime().freeMemory();
509                        if (free == free2) {
510                            break;
511                        }
512                    }
513                }
514                // don't wait too long so that deadlocks are detected early
515                long sleep = Math.min(Constants.DEADLOCK_CHECK, max - now);
516                if (sleep == 0) {
517                    sleep = 1;
518                }
519                waitForLock = true;
520                database.wait(sleep);
521            } catch (InterruptedException e) {
522                // ignore
523            }
524        }
525    }

527    private static String getDeadlockDetails(ArrayList<Session> sessions) {
528        StringBuilder buff = new StringBuilder();
529        for (Session s : sessions) {
530            Table lock = s.getWaitForLock();
531            buff.append("\nSession ").
532                append(s.toString()).
533                append(" is waiting to lock ").
534                append(lock.toString()).
535                append(" while locking ");
536            int i = 0;
537            for (Table t : s.getLocks()) {
538                if (i++ > 0) {
539                    buff.append(", ");
540                }
541                buff.append(t.toString());
542                if (t instanceof RegularTable) {
543                    if (((RegularTable) t).lockExclusive == s) {
544                        buff.append(" (exclusive)");
545                    } else {
546                        buff.append(" (shared)");
547                    }
548                }
549            }
550            buff.append('.');
551        }
552        return buff.toString();
553    }

555    public ArrayList<Session> checkDeadlock(Session session, Session clash, Set<Session> visited) {
556        // only one deadlock check at any given time
557        synchronized (RegularTable.class) {
558            if (clash == null) {
559                // verification is started
560                clash = session;
561                visited = New.hashSet();
562            } else if (clash == session) {
563                // we found a circle where this session is involved
564                return New.arrayList();
565            } else if (visited.contains(session)) {
566                // we have already checked this session.
567                // there is a circle, but the sessions in the circle need to
568                // find it out themselves
569                return null;
570            }
571            visited.add(session);
572            ArrayList<Session> error = null;
573            for (Session s : lockShared) {
574                if (s == session) {
575                    // it doesn't matter if we have locked the object already
576                    continue;
577                }
578                Table t = s.getWaitForLock();
579                if (t != null) {
580                    error = t.checkDeadlock(s, clash, visited);
581                    if (error != null) {
582                        error.add(session);
583                        break;
584                    }
585                }
586            }
587            if (error == null && lockExclusive != null) {
588                Table t = lockExclusive.getWaitForLock();
589                if (t != null) {
590                    error = t.checkDeadlock(lockExclusive, clash, visited);
591                    if (error != null) {
592                        error.add(session);
593                    }
594                }
595            }
596            return error;
597        }
598    }
