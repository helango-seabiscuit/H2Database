487    /**

489     * spec defines the first week of the year as the week which contains at

491     * December 29th - 31th could belong to the next year and January 1st - 3th

503        c.setTimeInMillis(date.getTime());
504        c.setFirstDayOfWeek(Calendar.MONDAY);
505        c.setMinimalDaysInFirstWeek(4);
506        return c.get(Calendar.WEEK_OF_YEAR);
507    }

591     * Verify if the specified date is valid.
592     *
593     * @param year the year
594     * @param month the month (January is 1)
595     * @param day the day (1 is the first of the month)
596     * @return true if it is valid
597     */
598    public static boolean isValidDate(int year, int month, int day) {
599        if (month < 1 || month > 12 || day < 1) {
600            return false;
601        }
602        if (year > 1582) {
603            // Gregorian calendar
604            if (month != 2) {
605                return day <= NORMAL_DAYS_PER_MONTH[month];
606            }
607            // February
608            if ((year & 3) != 0) {
609                return day <= 28;
610            }
611            return day <= ((year % 100 != 0) || (year % 400 == 0) ? 29 : 28);
612        } else if (year == 1582 && month == 10) {
613            // special case: days 1582-10-05 .. 1582-10-14 don't exist
614            return day <= 31 && (day < 5 || day > 14);
615        }
616        if (month != 2 && day <= NORMAL_DAYS_PER_MONTH[month]) {
617            return true;
618        }
619        return day <= ((year & 3) != 0 ? 28 : 29);
620    }

812    public static long absoluteDayFromDateValue(long dateValue) {

815        int d = dayFromDateValue(dateValue);

837    public static long dateValueFromAbsoluteDay(long absoluteDay) {
838        long d = absoluteDay + 719468;
839        long y100 = 0, offset;
840        if (d > 578040) {
841            // Gregorian calendar
842            long y400 = d / 146097;
843            d -= y400 * 146097;
844            y100 = d / 36524;
845            d -= y100 * 36524;
846            offset = y400 * 400 + y100 * 100;
847        } else {
848            // Julian calendar
849            d += 292200000002L;
850            offset = -800000000;
851        }
852        long y4 = d / 1461;
853        d -= y4 * 1461;
854        long y = d / 365;
855        d -= y * 365;

857            y--;
858            d += 365;
859        }
860        y += offset + y4 * 4;
861        // month of a day
862        int m = ((int) d * 2 + 1) * 5 / 306;
863        d -= DAYS_OFFSET[m] - 1;
864        if (m >= 10) {
865            y++;
866            m -= 12;
867        }
