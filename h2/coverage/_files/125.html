579    private boolean checkIndex(Session session, String value, Value indexFrom, Value indexTo) {
580        if (value == null || (indexFrom == null && indexTo == null)) {
581            return true;
582        }
583        Database db = session.getDatabase();
584        if (database.getMode().lowerCaseIdentifiers) {
585            Value v = ValueStringIgnoreCase.get(value);
586            if (indexFrom.equals(indexTo) && db.compare(v, indexFrom) != 0) {
587                return false;
588            }
589        } else {
590            Value v = ValueString.get(value);
591            if (indexFrom != null && db.compare(v, indexFrom) < 0) {
592                return false;
593            }
594            if (indexTo != null && db.compare(v, indexTo) > 0) {
595                return false;
596            }
597        }
598        return true;
