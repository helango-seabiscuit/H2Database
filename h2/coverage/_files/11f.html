449    /**
450     * Execute a SQL statement using the given parameters. Prepared
451     * statements are kept in a hash map to avoid re-creating them.
452     *
453     * @param sql the SQL statement
454     * @param params the parameters or null
455     * @param reusePrepared if the prepared statement can be re-used immediately
456     * @return the prepared statement, or null if it is re-used
457     */
458    public PreparedStatement execute(String sql, ArrayList<Value> params, boolean reusePrepared) {
459        if (conn == null) {
460            throw connectException;
461        }
462        for (int retry = 0;; retry++) {
463            try {
464                synchronized (conn) {
465                    PreparedStatement prep = preparedMap.remove(sql);
466                    if (prep == null) {
467                        prep = conn.getConnection().prepareStatement(sql);
468                    }
469                    if (trace.isDebugEnabled()) {
470                        StatementBuilder buff = new StatementBuilder();
471                        buff.append(getName()).append(":\n").append(sql);
472                        if (params != null && params.size() > 0) {
473                            buff.append(" {");
474                            int i = 1;
475                            for (Value v : params) {
476                                buff.appendExceptFirst(", ");
477                                buff.append(i++).append(": ").append(v.getSQL());
478                            }
479                            buff.append('}');
480                        }
481                        buff.append(';');
482                        trace.debug(buff.toString());
483                    }
484                    if (params != null) {
485                        for (int i = 0, size = params.size(); i < size; i++) {
486                            Value v = params.get(i);
487                            v.set(prep, i + 1);
488                        }
489                    }
490                    prep.execute();
491                    if (reusePrepared) {
492                        reusePreparedStatement(prep, sql);
493                        return null;
494                    }
495                    return prep;
496                }
497            } catch (SQLException e) {
498                if (retry >= MAX_RETRY) {
499                    throw DbException.convert(e);
500                }
501                conn.close(true);
502                connect();
503            }
504        }
505    }
