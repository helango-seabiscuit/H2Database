466    private void writeHead() {
467        data.reset();
468        int type;
469        if (firstOverflowPageId == 0) {
470            type = Page.TYPE_DATA_LEAF | Page.FLAG_LAST;
471        } else {
472            type = Page.TYPE_DATA_LEAF;
473        }
474        data.writeByte((byte) type);
475        data.writeShortInt(0);
476        if (SysProperties.CHECK2) {
477            if (data.length() != START_PARENT) {
478                DbException.throwInternalError();
479            }
480        }
481        data.writeInt(parentPageId);
482        data.writeVarInt(index.getId());
483        data.writeVarInt(columnCount);
484        data.writeShortInt(entryCount);
485    }

487    private void writeData() {
488        if (written) {
489            return;
490        }
491        if (!optimizeUpdate) {
492            readAllRows();
493        }
494        writeHead();
495        if (firstOverflowPageId != 0) {
496            data.writeInt(firstOverflowPageId);
497            data.checkCapacity(overflowRowSize);
498        }
499        for (int i = 0; i < entryCount; i++) {
500            data.writeVarLong(keys[i]);
501            data.writeShortInt(offsets[i]);
502        }
503        if (!writtenData || !optimizeUpdate) {
504            for (int i = 0; i < entryCount; i++) {
505                data.setPos(offsets[i]);
506                Row r = getRowAt(i);
507                for (int j = 0; j < columnCount; j++) {
508                    data.writeValue(r.getValue(j));
509                }
510            }
511            writtenData = true;
512        }
513        written = true;
514    }

523    public void moveTo(Session session, int newPos) {
524        PageStore store = index.getPageStore();
525        // load the pages into the cache, to ensure old pages
526        // are written
527        if (parentPageId != ROOT) {
528            store.getPage(parentPageId);
529        }
530        store.logUndo(this, data);
531        PageDataLeaf p2 = PageDataLeaf.create(index, newPos, parentPageId);
532        readAllRows();
533        p2.keys = keys;
534        p2.overflowRowSize = overflowRowSize;
535        p2.firstOverflowPageId = firstOverflowPageId;
536        p2.rowRef = rowRef;
537        p2.rows = rows;
538        if (firstOverflowPageId != 0) {
539            p2.rows[0] = getRowAt(0);
540        }
541        p2.entryCount = entryCount;
542        p2.offsets = offsets;
543        p2.start = start;
544        p2.remapChildren(getPos());
545        p2.writeData();
546        p2.data.truncate(index.getPageStore().getPageSize());
547        store.update(p2);
548        if (parentPageId == ROOT) {
549            index.setRootPageId(session, newPos);
550        } else {
551            PageDataNode p = (PageDataNode) store.getPage(parentPageId);
552            p.moveChild(getPos(), newPos);
553        }
554        store.free(getPos());
555    }
