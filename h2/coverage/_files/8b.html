443                writeByte((byte) type);
444                writeVarLong(x);
445            }
446            break;
447        }
448        case Value.DECIMAL: {
449            BigDecimal x = v.getBigDecimal();
450            if (BigDecimal.ZERO.equals(x)) {
451                writeByte((byte) DECIMAL_0_1);
452            } else if (BigDecimal.ONE.equals(x)) {
453                writeByte((byte) (DECIMAL_0_1 + 1));
454            } else {
455                int scale = x.scale();
456                BigInteger b = x.unscaledValue();
457                int bits = b.bitLength();
458                if (bits <= 63) {
459                    if (scale == 0) {
460                        writeByte((byte) DECIMAL_SMALL_0);
461                        writeVarLong(b.longValue());
462                    } else {
463                        writeByte((byte) DECIMAL_SMALL);
464                        writeVarInt(scale);
465                        writeVarLong(b.longValue());
466                    }
467                } else {
468                    writeByte((byte) type);
469                    writeVarInt(scale);
470                    byte[] bytes = b.toByteArray();
471                    writeVarInt(bytes.length);
472                    write(bytes, 0, bytes.length);
473                }
474            }
475            break;
476        }
477        case Value.TIME:
478            if (SysProperties.STORE_LOCAL_TIME) {
479                writeByte((byte) LOCAL_TIME);
480                ValueTime t = (ValueTime) v;
481                long nanos = t.getNanos();
482                long millis = nanos / 1000000;
483                nanos -= millis * 1000000;
484                writeVarLong(millis);
485                writeVarLong(nanos);
486            } else {
487                writeByte((byte) type);
488                writeVarLong(DateTimeUtils.getTimeLocalWithoutDst(v.getTime()));
489            }
490            break;
491        case Value.DATE: {
492            if (SysProperties.STORE_LOCAL_TIME) {
493                writeByte((byte) LOCAL_DATE);
494                long x = ((ValueDate) v).getDateValue();
495                writeVarLong(x);
496            } else {
497                writeByte((byte) type);
498                long x = DateTimeUtils.getTimeLocalWithoutDst(v.getDate());
499                writeVarLong(x / MILLIS_PER_MINUTE);
500            }
501            break;
502        }
503        case Value.TIMESTAMP: {
504            if (SysProperties.STORE_LOCAL_TIME) {
505                writeByte((byte) LOCAL_TIMESTAMP);
506                ValueTimestamp ts = (ValueTimestamp) v;
507                long dateValue = ts.getDateValue();
508                writeVarLong(dateValue);
509                long nanos = ts.getNanos();
510                long millis = nanos / 1000000;
511                nanos -= millis * 1000000;
512                writeVarLong(millis);
513                writeVarLong(nanos);
514            } else {
515                Timestamp ts = v.getTimestamp();
516                writeByte((byte) type);
517                writeVarLong(DateTimeUtils.getTimeLocalWithoutDst(ts));
518                writeVarInt(ts.getNanos());
519            }
520            break;
521        }
522        case Value.JAVA_OBJECT: {
523            writeByte((byte) type);
524            byte[] b = v.getBytesNoCopy();
525            writeVarInt(b.length);
526            write(b, 0, b.length);
527            break;
528        }
529        case Value.BYTES: {
530            byte[] b = v.getBytesNoCopy();
531            int len = b.length;
532            if (len < 32) {
533                writeByte((byte) (BYTES_0_31 + len));
534                write(b, 0, b.length);
535            } else {
536                writeByte((byte) type);
537                writeVarInt(b.length);
538                write(b, 0, b.length);
539            }
540            break;
541        }
542        case Value.UUID: {
543            writeByte((byte) type);
544            ValueUuid uuid = (ValueUuid) v;
545            writeLong(uuid.getHigh());
546            writeLong(uuid.getLow());
547            break;
548        }
549        case Value.STRING: {
550            String s = v.getString();
551            int len = s.length();
552            if (len < 32) {
553                writeByte((byte) (STRING_0_31 + len));
554                writeStringWithoutLength(s, len);
555            } else {
556                writeByte((byte) type);
557                writeString(s);
558            }
559            break;
560        }
561        case Value.STRING_IGNORECASE:
562        case Value.STRING_FIXED:
563            writeByte((byte) type);
564            writeString(v.getString());
565            break;
566        case Value.DOUBLE: {
567            double x = v.getDouble();
568            if (x == 1.0d) {
569                writeByte((byte) (DOUBLE_0_1 + 1));
570            } else {
571                long d = Double.doubleToLongBits(x);
572                if (d == ValueDouble.ZERO_BITS) {
573                    writeByte((byte) DOUBLE_0_1);
574                } else {
575                    writeByte((byte) type);
576                    writeVarLong(Long.reverse(d));
577                }
578            }
579            break;
580        }
581        case Value.FLOAT: {
582            float x = v.getFloat();
583            if (x == 1.0f) {
584                writeByte((byte) (FLOAT_0_1 + 1));
585            } else {
586                int f = Float.floatToIntBits(x);
587                if (f == ValueFloat.ZERO_BITS) {
588                    writeByte((byte) FLOAT_0_1);
589                } else {
590                    writeByte((byte) type);
591                    writeVarInt(Integer.reverse(f));
592                }
593            }
594            break;
595        }
596        case Value.BLOB:
597        case Value.CLOB: {
598            writeByte((byte) type);
599            if (v instanceof ValueLob) {
600                ValueLob lob = (ValueLob) v;
601                lob.convertToFileIfRequired(handler);
602                byte[] small = lob.getSmall();
603                if (small == null) {
604                    int t = -1;
605                    if (!lob.isLinked()) {
606                        t = -2;
607                    }
608                    writeVarInt(t);
609                    writeVarInt(lob.getTableId());
610                    writeVarInt(lob.getObjectId());
611                    writeVarLong(lob.getPrecision());
612                    writeByte((byte) (lob.useCompression() ? 1 : 0));
613                    if (t == -2) {
614                        writeString(lob.getFileName());
615                    }
616                } else {
617                    writeVarInt(small.length);
618                    write(small, 0, small.length);
619                }
620            } else {
621                ValueLobDb lob = (ValueLobDb) v;
622                byte[] small = lob.getSmall();
623                if (small == null) {
624                    writeVarInt(-3);
625                    writeVarInt(lob.getTableId());
626                    writeVarLong(lob.getLobId());
627                    writeVarLong(lob.getPrecision());
628                } else {
629                    writeVarInt(small.length);
630                    write(small, 0, small.length);
631                }
632            }
633            break;
634        }
635        case Value.ARRAY: {
636            writeByte((byte) type);
637            Value[] list = ((ValueArray) v).getList();
638            writeVarInt(list.length);
639            for (Value x : list) {
640                writeValue(x);
641            }
642            break;
643        }
644        case Value.RESULT_SET: {
645            writeByte((byte) type);
646            try {
647                ResultSet rs = ((ValueResultSet) v).getResultSet();
648                rs.beforeFirst();
649                ResultSetMetaData meta = rs.getMetaData();
650                int columnCount = meta.getColumnCount();
651                writeVarInt(columnCount);
652                for (int i = 0; i < columnCount; i++) {
653                    writeString(meta.getColumnName(i + 1));
654                    writeVarInt(meta.getColumnType(i + 1));
655                    writeVarInt(meta.getPrecision(i + 1));
656                    writeVarInt(meta.getScale(i + 1));
657                }
658                while (rs.next()) {
659                    writeByte((byte) 1);
660                    for (int i = 0; i < columnCount; i++) {
661                        int t = DataType.convertSQLTypeToValueType(meta.getColumnType(i + 1));
662                        Value val = DataType.readValue(null, rs, i + 1, t);
663                        writeValue(val);
664                    }
665                }
666                writeByte((byte) 0);
667                rs.beforeFirst();
668            } catch (SQLException e) {
669                throw DbException.convert(e);
670            }
671            break;
672        }
673        default:
674            DbException.throwInternalError("type=" + v.getType());
675        }
676        if (SysProperties.CHECK2) {
677            if (pos - start != getValueLen(v, handler)) {
678                throw DbException
679                        .throwInternalError("value size error: got " + (pos - start) + " expected " + getValueLen(v, handler));
680            }
681        }
682    }

687     * @return the value
688     */

691        switch (type) {
692        case Value.NULL:
693            return ValueNull.INSTANCE;
694        case BOOLEAN_TRUE:
695            return ValueBoolean.get(true);
696        case BOOLEAN_FALSE:
697            return ValueBoolean.get(false);
698        case INT_NEG:
699            return ValueInt.get(-readVarInt());
700        case Value.INT:
701            return ValueInt.get(readVarInt());
702        case LONG_NEG:
703            return ValueLong.get(-readVarLong());
704        case Value.LONG:
705            return ValueLong.get(readVarLong());
706        case Value.BYTE:
707            return ValueByte.get(readByte());
708        case Value.SHORT:
709            return ValueShort.get(readShortInt());
710        case DECIMAL_0_1:
711            return (ValueDecimal) ValueDecimal.ZERO;
712        case DECIMAL_0_1 + 1:
713            return (ValueDecimal) ValueDecimal.ONE;
714        case DECIMAL_SMALL_0:
715            return ValueDecimal.get(BigDecimal.valueOf(readVarLong()));
716        case DECIMAL_SMALL: {
717            int scale = readVarInt();
718            return ValueDecimal.get(BigDecimal.valueOf(readVarLong(), scale));
719        }
720        case Value.DECIMAL: {
721            int scale = readVarInt();
722            int len = readVarInt();
723            byte[] buff = Utils.newBytes(len);
724            read(buff, 0, len);
725            BigInteger b = new BigInteger(buff);
726            return ValueDecimal.get(new BigDecimal(b, scale));
727        }
728        case LOCAL_DATE: {
729            return ValueDate.fromDateValue(readVarLong());
730        }
731        case Value.DATE: {
732            long x = readVarLong() * MILLIS_PER_MINUTE;
733            return ValueDate.get(new Date(DateTimeUtils.getTimeUTCWithoutDst(x)));
734        }
735        case LOCAL_TIME: {
736            long nanos = readVarLong() * 1000000 + readVarLong();
737            return ValueTime.fromNanos(nanos);
738        }
739        case Value.TIME:
740            // need to normalize the year, month and day
741            return ValueTime.get(new Time(DateTimeUtils.getTimeUTCWithoutDst(readVarLong())));
742        case LOCAL_TIMESTAMP: {
743            long dateValue = readVarLong();
744            long nanos = readVarLong() * 1000000 + readVarLong();
745            return ValueTimestamp.fromDateValueAndNanos(dateValue, nanos);
746        }
747        case Value.TIMESTAMP: {
748            Timestamp ts = new Timestamp(DateTimeUtils.getTimeUTCWithoutDst(readVarLong()));
749            ts.setNanos(readVarInt());
750            return ValueTimestamp.get(ts);
751        }
752        case Value.BYTES: {
753            int len = readVarInt();
754            byte[] b = Utils.newBytes(len);
755            read(b, 0, len);
756            return ValueBytes.getNoCopy(b);
757        }
758        case Value.JAVA_OBJECT: {
759            int len = readVarInt();
760            byte[] b = Utils.newBytes(len);
761            read(b, 0, len);
762            return ValueJavaObject.getNoCopy(null, b);
763        }
764        case Value.UUID:
765            return ValueUuid.get(readLong(), readLong());
766        case Value.STRING:
767            return ValueString.get(readString());
768        case Value.STRING_IGNORECASE:
769            return ValueStringIgnoreCase.get(readString());
770        case Value.STRING_FIXED:
771            return ValueStringFixed.get(readString());
772        case FLOAT_0_1:
773            return ValueFloat.get(0);
774        case FLOAT_0_1 + 1:
775            return ValueFloat.get(1);
776        case DOUBLE_0_1:
777            return ValueDouble.get(0);
778        case DOUBLE_0_1 + 1:
779            return ValueDouble.get(1);
780        case Value.DOUBLE:
781            return ValueDouble.get(Double.longBitsToDouble(Long.reverse(readVarLong())));
782        case Value.FLOAT:
783            return ValueFloat.get(Float.intBitsToFloat(Integer.reverse(readVarInt())));
784        case Value.BLOB:
785        case Value.CLOB: {
786            int smallLen = readVarInt();
787            if (smallLen >= 0) {
788                byte[] small = Utils.newBytes(smallLen);
789                read(small, 0, smallLen);
790                return LobStorage.createSmallLob(type, small);
791            } else if (smallLen == -3) {
792                int tableId = readVarInt();
793                long lobId = readVarLong();
794                long precision = readVarLong();
795                LobStorage lobStorage = handler.getLobStorage();
796                ValueLobDb lob = ValueLobDb.create(type, lobStorage, tableId, lobId, null, precision);
797                return lob;
798            } else {
799                int tableId = readVarInt();
800                int objectId = readVarInt();
801                long precision = 0;
802                boolean compression = false;
803                // -1: regular
804                // -2: regular, but not linked (in this case: including file name)
805                if (smallLen == -1 || smallLen == -2) {
806                    precision = readVarLong();
807                    compression = readByte() == 1;
808                }
809                ValueLob lob = ValueLob.open(type, handler, tableId, objectId, precision, compression);
810                if (smallLen == -2) {
811                    lob.setFileName(readString(), false);
812                }
813                return lob;
814            }
815        }
816        case Value.ARRAY: {
817            int len = readVarInt();
818            Value[] list = new Value[len];
819            for (int i = 0; i < len; i++) {
820                list[i] = readValue();
821            }
822            return ValueArray.get(list);
823        }
824        case Value.RESULT_SET: {
825            SimpleResultSet rs = new SimpleResultSet();
826            int columns = readVarInt();
827            for (int i = 0; i < columns; i++) {
828                rs.addColumn(readString(), readVarInt(), readVarInt(), readVarInt());
829            }
830            while (true) {
831                if (readByte() == 0) {
832                    break;
833                }
834                Object[] o = new Object[columns];
835                for (int i = 0; i < columns; i++) {
836                    o[i] = readValue().getObject();
837                }
838                rs.addRow(o);
839            }
840            return ValueResultSet.get(rs);
841        }
842        default:
843            if (type >= INT_0_15 && type < INT_0_15 + 16) {
844                return ValueInt.get(type - INT_0_15);
845            } else if (type >= LONG_0_7 && type < LONG_0_7 + 8) {
846                return ValueLong.get(type - LONG_0_7);
847            } else if (type >= BYTES_0_31 && type < BYTES_0_31 + 32) {
848                int len = type - BYTES_0_31;
849                byte[] b = Utils.newBytes(len);
850                read(b, 0, len);
851                return ValueBytes.getNoCopy(b);
852            } else if (type >= STRING_0_31 && type < STRING_0_31 + 32) {
853                return ValueString.get(readString(type - STRING_0_31));
854            }
855            throw DbException.get(ErrorCode.FILE_CORRUPTED_1, "type: " + type);
856        }

869    /**
870     * Calculate the number of bytes required to encode the given value.
871     *
872     * @param v the value
873     * @param handler the data handler for lobs
874     * @return the number of bytes required to store this value
875     */
876    private static int getValueLen(Value v, DataHandler handler) {
877        if (v == ValueNull.INSTANCE) {
878            return 1;
879        }
880        switch (v.getType()) {
881        case Value.BOOLEAN:
882            return 1;
883        case Value.BYTE:
884            return 2;
885        case Value.SHORT:
886            return 3;
887        case Value.INT: {
888            int x = v.getInt();
889            if (x < 0) {
890                return 1 + getVarIntLen(-x);
891            } else if (x < 16) {
892                return 1;
893            } else {
894                return 1 + getVarIntLen(x);
895            }
896        }
897        case Value.LONG: {
898            long x = v.getLong();
899            if (x < 0) {
900                return 1 + getVarLongLen(-x);
901            } else if (x < 8) {
902                return 1;
903            } else {
904                return 1 + getVarLongLen(x);
905            }
906        }
907        case Value.DOUBLE: {
908            double x = v.getDouble();
909            if (x == 1.0d) {
910                return 1;
911            }
912            long d = Double.doubleToLongBits(x);
913            if (d == ValueDouble.ZERO_BITS) {
914                return 1;
915            }
916            return 1 + getVarLongLen(Long.reverse(d));
917        }
918        case Value.FLOAT: {
919            float x = v.getFloat();
920            if (x == 1.0f) {
921                return 1;
922            }
923            int f = Float.floatToIntBits(x);
924            if (f == ValueFloat.ZERO_BITS) {
925                return 1;
926            }
927            return 1 + getVarIntLen(Integer.reverse(f));
928        }
929        case Value.STRING: {
930            String s = v.getString();
931            int len = s.length();
932            if (len < 32) {
933                return 1 + getStringWithoutLengthLen(s, len);
934            }
935            return 1 + getStringLen(s);
936        }
937        case Value.STRING_IGNORECASE:
938        case Value.STRING_FIXED:
939            return 1 + getStringLen(v.getString());
940        case Value.DECIMAL: {
941            BigDecimal x = v.getBigDecimal();
942            if (BigDecimal.ZERO.equals(x)) {
943                return 1;
944            } else if (BigDecimal.ONE.equals(x)) {
945                return 1;
946            }
947            int scale = x.scale();
948            BigInteger b = x.unscaledValue();
