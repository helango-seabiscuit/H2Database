449     * @return the server if successful
450     * @throws SQLException if the server could not be started
451     */
452    public Server start() throws SQLException {
453        try {
454            started = true;
455            service.start();
456            String name = service.getName() + " (" + service.getURL() + ")";
457            Thread t = new Thread(this, name);
458            t.setDaemon(service.isDaemon());
459            t.start();
460            for (int i = 1; i < 64; i += i) {
461                wait(i);
462                if (isRunning(false)) {
463                    return this;
464                }
465            }
466            if (isRunning(true)) {
467                return this;
468            }
469            throw DbException.get(ErrorCode.EXCEPTION_OPENING_PORT_2, name, "timeout; " +
470                    "please check your network configuration, specially the file /etc/hosts");
471        } catch (DbException e) {
472            throw DbException.toSQLException(e);
473        }
474    }

577    /**
578     * Open a new browser tab or window with the given URL.
579     *
580     * @param url the URL to open
581     */
582    public static void openBrowser(String url) throws Exception {
583        try {
584            String osName = StringUtils.toLowerEnglish(Utils.getProperty("os.name", "linux"));
585            Runtime rt = Runtime.getRuntime();
586            String browser = Utils.getProperty(SysProperties.H2_BROWSER, null);
587            if (browser != null) {
588                if (browser.startsWith("call:")) {
589                    browser = browser.substring("call:".length());
590                    Utils.callStaticMethod(browser, url);
591                } else if (browser.indexOf("%url") >= 0) {
592                    String[] args = StringUtils.arraySplit(browser, ',', false);
593                    for (int i = 0; i < args.length; i++) {
594                        args[i] = StringUtils.replaceAll(args[i], "%url", url);
595                    }
596                    rt.exec(args);
597                } else if (osName.indexOf("windows") >= 0) {
598                    rt.exec(new String[] { "cmd.exe", "/C",  browser, url });
599                } else {
600                    rt.exec(new String[] { browser, url });
601                }
602                return;
603            }
604            try {
605                Class<?> desktopClass = Class.forName("java.awt.Desktop");
606                // Desktop.isDesktopSupported()
607                Boolean supported = (Boolean) desktopClass.
608                    getMethod("isDesktopSupported").
609                    invoke(null, new Object[0]);
610                URI uri = new URI(url);
611                if (supported) {
612                    // Desktop.getDesktop();
613                    Object desktop = desktopClass.getMethod("getDesktop").
614                        invoke(null, new Object[0]);
615                    // desktop.browse(uri);
616                    desktopClass.getMethod("browse", URI.class).
617                        invoke(desktop, uri);
618                    return;
619                }
620            } catch (Exception e) {
621                // ignore
622            }
623            if (osName.indexOf("windows") >= 0) {
624                rt.exec(new String[] { "rundll32", "url.dll,FileProtocolHandler", url });
625            } else if (osName.indexOf("mac") >= 0 || osName.indexOf("darwin") >= 0) {
626                // Mac OS: to open a page with Safari, use "open -a Safari"
627                Runtime.getRuntime().exec(new String[] { "open", url });
628            } else {
629                String[] browsers = { "google-chrome", "firefox", "mozilla-firefox",
630                        "mozilla", "konqueror", "netscape", "opera", "midori" };
631                boolean ok = false;
632                for (String b : browsers) {
633                    try {
634                        rt.exec(new String[] { b, url });
635                        ok = true;
636                        break;
637                    } catch (Exception e) {
638                        // ignore and try the next
639                    }
640                }
641                if (!ok) {
642                    // No success in detection.
643                    throw new Exception("Browser detection failed and system property " + SysProperties.H2_BROWSER + " not set");
644                }
645            }
646        } catch (Exception e) {
647            throw new Exception("Failed to start a browser to open the URL " + url + ": " + e.getMessage());
648        }
649    }

653     * current transaction is preserved. This is specially useful to manually
