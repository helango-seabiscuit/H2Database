436                buff = IOUtils.readBytesAndClose(in, -1);
437                len = buff.length;
438            } else {
439                buff = Utils.newBytes(len);
440                len = IOUtils.readFully(in, buff, 0, len);
441            }
442            if (len <= handler.getMaxLengthInplaceLob()) {
443                byte[] small = Utils.newBytes(len);
444                System.arraycopy(buff, 0, small, 0, len);
445                return ValueLobDb.createSmallLob(Value.BLOB, small, small.length);
446            }
447            ValueLobDb lob = new ValueLobDb(Value.BLOB, null, 0);
448            lob.createTempFromStream(buff, len, in, remaining, handler);
449            return lob;
450        } catch (IOException e) {

452        }

527    private static int getBufferSize(DataHandler handler, boolean compress, long remaining) {
528        if (remaining < 0 || remaining > Integer.MAX_VALUE) {
529            remaining = Integer.MAX_VALUE;
530        }
531        long inplace = handler.getMaxLengthInplaceLob();
532        if (inplace >= Integer.MAX_VALUE) {
533            inplace = remaining;
534        }
535        long m = compress ? Constants.IO_BUFFER_SIZE_COMPRESS : Constants.IO_BUFFER_SIZE;
536        if (m < remaining && m <= inplace) {
537            m = Math.min(remaining, inplace + 1);
538            // the buffer size must be bigger than the inplace lob, otherwise we can't
539            // know if it must be stored in-place or not
540            m = MathUtils.roundUpLong(m, Constants.IO_BUFFER_SIZE);
541        }
542        m = Math.min(remaining, m);
543        m = MathUtils.convertLongToInt(m);
544        if (m < 0) {
545            m = Integer.MAX_VALUE;
546        }
547        return (int) m;
548    }
