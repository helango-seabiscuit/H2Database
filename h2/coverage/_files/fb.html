281    private Prepared parsePrepared() {
282        int start = lastParseIndex;
283        Prepared c = null;
284        String token = currentToken;
285        if (token.length() == 0) {
286            c = new NoOperation(session);
287        } else {
288            char first = token.charAt(0);
289            switch (first) {
290            case '?':
291                // read the ? as a parameter
292                readTerm();
293                // this is an 'out' parameter - set a dummy value
294                parameters.get(0).setValue(ValueNull.INSTANCE);
295                read("=");
296                read("CALL");
297                c = parseCall();
298                break;
299            case '(':
300                c = parseSelect();
301                break;
302            case 'a':
303            case 'A':
304                if (readIf("ALTER")) {
305                    c = parseAlter();
306                } else if (readIf("ANALYZE")) {
307                    c = parseAnalyze();
308                }
309                break;
310            case 'b':
311            case 'B':
312                if (readIf("BACKUP")) {
313                    c = parseBackup();
314                } else if (readIf("BEGIN")) {
315                    c = parseBegin();
316                }
317                break;
318            case 'c':
319            case 'C':
320                if (readIf("COMMIT")) {
321                    c = parseCommit();
322                } else if (readIf("CREATE")) {
323                    c = parseCreate();
324                } else if (readIf("CALL")) {
325                    c = parseCall();
326                } else if (readIf("CHECKPOINT")) {
327                    c = parseCheckpoint();
328                } else if (readIf("COMMENT")) {
329                    c = parseComment();
330                }
331                break;
332            case 'd':
333            case 'D':
334                if (readIf("DELETE")) {
335                    c = parseDelete();
336                } else if (readIf("DROP")) {
337                    c = parseDrop();
338                } else if (readIf("DECLARE")) {
339                    // support for DECLARE GLOBAL TEMPORARY TABLE...
340                    c = parseCreate();
341                } else if (readIf("DEALLOCATE")) {
342                    c = parseDeallocate();
343                }
344                break;
345            case 'e':
346            case 'E':
347                if (readIf("EXPLAIN")) {
348                    c = parseExplain();
349                } else if (readIf("EXECUTE")) {
350                    c = parseExecute();
351                }
352                break;
353            case 'f':
354            case 'F':
355                if (isToken("FROM")) {
356                    c = parseSelect();
357                }
358                break;
359            case 'g':
360            case 'G':
361                if (readIf("GRANT")) {
362                    c = parseGrantRevoke(CommandInterface.GRANT);
363                }
364                break;
365            case 'h':
366            case 'H':
367                if (readIf("HELP")) {
368                    c = parseHelp();
369                }
370                break;
371            case 'i':
372            case 'I':
373                if (readIf("INSERT")) {
374                    c = parseInsert();
375                }
376                break;
377            case 'm':
378            case 'M':
379                if (readIf("MERGE")) {
380                    c = parseMerge();
381                }
382                break;
383            case 'p':
384            case 'P':
385                if (readIf("PREPARE")) {
386                    c = parsePrepare();
387                }
388                break;
389            case 'r':
390            case 'R':
391                if (readIf("ROLLBACK")) {
392                    c = parseRollback();
393                } else if (readIf("REVOKE")) {
394                    c = parseGrantRevoke(CommandInterface.REVOKE);
395                } else if (readIf("RUNSCRIPT")) {
396                    c = parseRunScript();
397                } else if (readIf("RELEASE")) {
398                    c = parseReleaseSavepoint();
399                }
400                break;
401            case 's':
402            case 'S':
403                if (isToken("SELECT")) {
404                    c = parseSelect();
405                } else if (readIf("SET")) {
406                    c = parseSet();
407                } else if (readIf("SAVEPOINT")) {
408                    c = parseSavepoint();
409                } else if (readIf("SCRIPT")) {
410                    c = parseScript();
411                } else if (readIf("SHUTDOWN")) {
412                    c = parseShutdown();
413                } else if (readIf("SHOW")) {
414                    c = parseShow();
415                }
416                break;
417            case 't':
418            case 'T':
419                if (readIf("TRUNCATE")) {
420                    c = parseTruncate();
421                }
422                break;
423            case 'u':
424            case 'U':
425                if (readIf("UPDATE")) {
426                    c = parseUpdate();
427                }
428                break;
429            case 'v':
430            case 'V':
431                if (readIf("VALUES")) {
432                    c = parseValues();
433                }
434                break;
435            case 'w':
436            case 'W':
437                if (readIf("WITH")) {
438                    c = parserWith();
439                }
440                break;
441            default:
442                throw getSyntaxError();
443            }
444            if (indexedParameterList != null) {
445                for (int i = 0, size = indexedParameterList.size(); i < size; i++) {
446                    if (indexedParameterList.get(i) == null) {
447                        indexedParameterList.set(i, new Parameter(i));
448                    }
449                }
450                parameters = indexedParameterList;
451            }
452            if (readIf("{")) {
453                do {
454                    int index = (int) readLong() - 1;
455                    if (index < 0 || index >= parameters.size()) {
456                        throw getSyntaxError();
457                    }
458                    Parameter p = parameters.get(index);
459                    if (p == null) {
460                        throw getSyntaxError();
461                    }
462                    read(":");
463                    Expression expr = readExpression();
464                    expr = expr.optimize(session);
465                    p.setValue(expr.getValue(session));
466                } while (readIf(","));
467                read("}");
468                for (Parameter p : parameters) {
469                    p.checkSet();
470                }
471                parameters.clear();
472            }
473        }
474        if (c == null) {
475            throw getSyntaxError();
476        }
477        setSQL(c, null, start);
478        return c;
479    }

481    private DbException getSyntaxError() {
482        if (expectedList == null || expectedList.size() == 0) {

488            buff.append(e);
489        }
490        return DbException.getSyntaxError(sqlCommand, parseIndex, buff.toString());
491    }

631                columnName = readColumnIdentifier();
632                if (readIf(".")) {
633                    String catalogName = schema;
634                    schema = tableAlias;
635                    tableAlias = columnName;

637                    if (!equalsToken(catalogName, database.getShortName())) {

657    private Update parseUpdate() {
658        Update command = new Update(session);
659        currentPrepared = command;
660        int start = lastParseIndex;
661        TableFilter filter = readSimpleTableFilter();
662        command.setTableFilter(filter);
663        read("SET");
664        if (readIf("(")) {
665            ArrayList<Column> columns = New.arrayList();
666            do {
667                Column column = readTableColumn(filter);
668                columns.add(column);
669            } while (readIf(","));
670            read(")");
671            read("=");
672            Expression expression = readExpression();
673            if (columns.size() == 1) {
674                // the expression is parsed as a simple value
675                command.setAssignment(columns.get(0), expression);
676            } else {
677                for (int i = 0, size = columns.size(); i < size; i++) {
678                    Column column = columns.get(i);
679                    Function f = Function.getFunction(database, "ARRAY_GET");
680                    f.setParameter(0, expression);
681                    f.setParameter(1, ValueExpression.get(ValueInt.get(i + 1)));
682                    f.doneWithParameters();
683                    command.setAssignment(column, f);
