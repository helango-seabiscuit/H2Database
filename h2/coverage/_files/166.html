425            // on an update, if both old and new are the same, don't do anything
426            return;
427        }
428        if (newRow == null) {
429            // this is a delete
430            if (deleteAction == RESTRICT) {
431                checkRow(session, oldRow);
432            } else {
433                int i = deleteAction == CASCADE ? 0 : columns.length;
434                Prepared deleteCommand = getDelete(session);
435                setWhere(deleteCommand, i, oldRow);
436                updateWithSkipCheck(deleteCommand);
437            }
438        } else {
439            // this is an update
440            if (updateAction == RESTRICT) {
441                checkRow(session, oldRow);
442            } else {
443                Prepared updateCommand = getUpdate(session);
444                if (updateAction == CASCADE) {
445                    ArrayList<Parameter> params = updateCommand.getParameters();
446                    for (int i = 0, len = columns.length; i < len; i++) {
447                        Parameter param = params.get(i);
448                        Column refCol = refColumns[i].column;
449                        param.setValue(newRow.getValue(refCol.getColumnId()));
450                    }
451                }
452                setWhere(updateCommand, columns.length, oldRow);
453                updateWithSkipCheck(updateCommand);
454            }
455        }
456    }

558    private Prepared prepare(Session session, String sql, int action) {
559        Prepared command = session.prepare(sql);
560        if (action != CASCADE) {
561            ArrayList<Parameter> params = command.getParameters();
562            for (int i = 0, len = columns.length; i < len; i++) {
563                Column column = columns[i].column;
564                Parameter param = params.get(i);
565                Value value;
566                if (action == SET_NULL) {
567                    value = ValueNull.INSTANCE;
568                } else {
569                    Expression expr = column.getDefaultExpression();
570                    if (expr == null) {
571                        throw DbException.get(ErrorCode.NO_DEFAULT_SET_1, column.getName());
572                    }
573                    value = expr.getValue(session);
574                }
575                param.setValue(value);
576            }
577        }
578        return command;
579    }
