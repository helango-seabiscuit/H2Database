477    void addUndo(int pageId, Data page) {
478        if (undo.get(pageId) || freeing) {

482            trace.debug("log undo " + pageId);
483        }

501                    buffer.writeVarInt(size);
502                    buffer.checkCapacity(size);
503                    buffer.write(compressBuffer, 0, size);
504                } else {
505                    buffer.writeVarInt(0);
506                    buffer.checkCapacity(pageSize);
507                    buffer.write(page.getBytes(), 0, pageSize);
508                }
509            } else {
510                buffer.writeVarInt(0);
511                buffer.checkCapacity(pageSize);
512                buffer.write(page.getBytes(), 0, pageSize);
513            }
514        }

720     * @param firstDataPageToKeep the first data page to keep
721     * @return the trunk page of the data page to keep
722     */
723    private int removeUntil(int trunkPage, int firstDataPageToKeep) {
724        trace.debug("log.removeUntil " + trunkPage + " " + firstDataPageToKeep);
